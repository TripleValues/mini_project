-- feat01 NULL 확인

SELECT *
FROM feat_01
WHERE 
    년도 IS NULL
    OR 월 IS NULL
    OR 승차인원합계 IS NULL
    OR 하차인원합계 IS NULL;
    
-- feat01 특정날짜/월 데이터 없나확인

SELECT f.년도, f.월
FROM (
    SELECT 2026 AS 년도, 1 AS 월 UNION ALL
    SELECT 2026, 2 UNION ALL
    SELECT 2026, 3 UNION ALL
    SELECT 2026, 4 UNION ALL
    SELECT 2026, 5 UNION ALL
    SELECT 2026, 6 UNION ALL
    SELECT 2026, 7 UNION ALL
    SELECT 2026, 8 UNION ALL
    SELECT 2026, 9 UNION ALL
    SELECT 2026, 10 UNION ALL
    SELECT 2026, 11 UNION ALL
    SELECT 2026, 12
) base
LEFT JOIN feat_01 f
    ON base.년도 = f.년도
   AND base.월 = f.월
WHERE f.년도 IS NULL;

-- feat01 중복체크

SELECT 년도, 월, COUNT(*) AS cnt
FROM feat_01
GROUP BY 년도, 월
HAVING COUNT(*) > 1;

-- feat01 종합체크쿼리

SELECT 
    COUNT(*) AS 전체건수,
    SUM(CASE WHEN 승차인원합계 IS NULL THEN 1 ELSE 0 END) AS 승차_NULL,
    SUM(CASE WHEN 하차인원합계 IS NULL THEN 1 ELSE 0 END) AS 하차_NULL,
    SUM(CASE WHEN 승차인원합계 = 0 THEN 1 ELSE 0 END) AS 승차_0,
    SUM(CASE WHEN 하차인원합계 = 0 THEN 1 ELSE 0 END) AS 하차_0
FROM feat_01;

-- feat02 null체크
SELECT *
FROM feat_02
WHERE 
    날짜 IS NULL
    OR 역명 IS NULL
    OR 시간대 IS NULL
    OR 기준 IS NULL
    OR 인원 IS NULL
    OR 순위 IS NULL;
    
-- feat02 중복데이터 체크
SELECT 
    날짜, 역명, 시간대, 기준,
    COUNT(*) AS cnt
FROM feat_02
GROUP BY 
    날짜, 역명, 시간대, 기준
HAVING COUNT(*) > 1;

-- feat02 순위 중복체크
SELECT 
    날짜, 시간대, 기준, 순위,
    COUNT(*) AS cnt
FROM feat_02
GROUP BY 
    날짜, 시간대, 기준, 순위
HAVING COUNT(*) > 1;

-- feat02 순위 누락 체크
SELECT 
    날짜, 시간대, 기준,
    MIN(순위) AS min_rank,
    MAX(순위) AS max_rank,
    COUNT(*) AS cnt
FROM feat_02
GROUP BY 날짜, 시간대, 기준
HAVING 
    MIN(순위) != 1
    OR MAX(순위) != 50
    OR COUNT(*) != 50;
    
-- feat02 시간대 값 이상 체크
SELECT DISTINCT 시간대
FROM feat_02
WHERE 시간대 NOT IN (
    '05~06','06~07','07~08','08~09','09~10',
    '10~11','11~12','12~13','13~14','14~15',
    '15~16','16~17','17~18','18~19','19~20',
    '20~21','21~22','22~23','23~24','24~',
    'ALL'
);

-- feat02 기준 값 검증
SELECT DISTINCT 기준
FROM feat_02
WHERE 기준 NOT IN ('승차','하차');

-- feat02 인원 vs 순위 검증 (논리 오류 체크🔥)
SELECT *
FROM (
    SELECT *,
        LAG(인원) OVER (
            PARTITION BY 날짜, 시간대, 기준
            ORDER BY 순위
        ) AS prev_value
    FROM feat_02
) t
WHERE 인원 > prev_value;

-- feat02 일일(ALL) 데이터 존재 체크
SELECT 날짜, 기준
FROM feat_02
WHERE 시간대 = 'ALL'
GROUP BY 날짜, 기준
HAVING COUNT(*) != 50;

-- feat02 종합 요약 리포트 (추천)
SELECT 
    COUNT(*) AS 전체건수,

    SUM(CASE WHEN 인원 < 0 THEN 1 ELSE 0 END) AS 이상_인원,
    SUM(CASE WHEN 순위 <= 0 THEN 1 ELSE 0 END) AS 이상_순위,

    COUNT(DISTINCT 날짜) AS 날짜수,
    COUNT(DISTINCT 시간대) AS 시간대수,
    COUNT(DISTINCT 기준) AS 기준수

FROM feat_02;

-- 전체건수 = 날짜수 × 시간대수 × 기준수 × 50가 동일해야함
